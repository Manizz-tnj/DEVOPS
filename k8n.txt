1.apt update
2.apt install -y apt-transport-https ca-certificates curl gpg
3.mkdir -p /etc/apt/keyrings
4.curl -fsSL https://pkgs.k8s.io/core:/stable:/v1.29/deb/Release.key | gpg --dearmor -o /etc/apt/keyrings/kubernetes-apt-keyring.gpg
5.echo "deb [signed-by=/etc/apt/keyrings/kubernetes-apt-keyring.gpg] https://pkgs.k8s.io/core:/stable:/v1.29/deb/ /" | tee /etc/apt/sources.list.d/kubernetes.list
6.sudo apt install -y containerd
7.sudo mkdir -p /etc/containerd
8.sudo containerd config default | sudo tee /etc/containerd/config.toml
9.sudo systemctl restart containerd
10.sudo systemctl enable containerd
11.apt update
12.apt install -y kubelet kubeadm kubectl
13.apt-mark hold kubelet kubeadm kubectl
14.sudo swapoff -a
15.free -m
